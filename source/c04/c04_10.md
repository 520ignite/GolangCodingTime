# 4.10 如何实现一个协程池？

协程池的实现

```go
type Pool struct {
  work chan func() // 任务  sem  chan struct{} // 数量
}
func New(size int) *Pool {
    return &Pool{
        work: make(chan func()),
        sem:  make(chan struct{}, size),
    }
}
func (p *Pool) Schedule(task func()) error { // 任务调度
    select {
    case p.work <- task:
    case p.sem <- struct{}{}:
        go p.worker(task)
    }
}
func (p *Pool) worker(task func()) { // 任务执行
    defer func() { <-p.sem }
    for {
        task()
        task = <-p.work
    }
}

```
